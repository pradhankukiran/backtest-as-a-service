Submit a backtest from the shell

# 1. ingest bars for a symbol python manage.py ingest_bars AAPL --create-missing --days-back 365 # 2. install the built-in SMA crossover strategy python manage.py install_builtin_strategies # 3. POST a run via the REST API (or use the admin) curl -X POST http://localhost:8000/api/runs/ \ -H 'Content-Type: application/json' \ -d '{ "strategy": "sma-crossover", "symbols": ["AAPL"], "start_date": "2024-04-01", "end_date": "2025-04-01", "initial_capital": 10000, "commission_bps": 20, "params": {"sma_short": 10, "sma_long": 30} }'
